import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
import matplotlib

# Define NBA top 10 players and their statistics
data = {
    "Player": [
        "Michael Jordan", "LeBron James", "Kareem Abdul-Jabbar", "Magic Johnson", "Kobe Bryant",
        "Shaquille O'Neal", "Tim Duncan", "Larry Bird", "Bill Russell", "Wilt Chamberlain"
    ],
    "Total Points": [
        32292, 40000, 38387, 17707, 33643,
        28596, 26496, 21791, 14522, 31419
    ],
    "Total Rebounds": [
        6672, 10818, 17440, 6559, 7047,
        13099, 15091, 8974, 21620, 23924
    ],
    "Total Assists": [
        5633, 10859, 5660, 10141, 6306,
        3026, 4225, 5695, 4100, 4643
    ],
    "Total Blocks": [
        893, 1085, 3189, 374, 640,
        2732, 3020, 755, 822, 3286
    ],
    "Total Steals": [
        2514, 2245, 1120, 1724, 1944,
        739, 1014, 1556, 0, 0
    ],
    "Championships": [
        6, 4, 6, 5, 5,
        4, 5, 3, 11, 2
    ],
    "MVPs": [
        5, 4, 6, 3, 1,
        1, 2, 3, 5, 4
    ]
}

# Create DataFrame
df = pd.DataFrame(data)

# Set categories for the bar chart
categories = ["Total Points", "Total Rebounds", "Total Assists", "Total Blocks", "Total Steals", "Championships", "MVPs"]
x = np.arange(len(df["Player"]))  # X-axis positions
width = 0.12  # Bar width

# Create a bar chart
fig, ax = plt.subplots(figsize=(14, 7))

# Plot bars for each category
for i, category in enumerate(categories):
    ax.bar(x + i * width, df[category], width, label=category)

# Configure X-axis
ax.set_xlabel("Player", fontsize=12)
ax.set_ylabel("Value", fontsize=12)
ax.set_title("NBA Top 10 Players Statistics Comparison", fontsize=14)
ax.set_xticks(x + width * (len(categories) / 2))
ax.set_xticklabels(df["Player"], rotation=45)
ax.legend()

# Show plot
plt.show()
